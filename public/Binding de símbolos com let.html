Binding de símbolos com let

https://clojuredocs.org/clojure.core/let

;; Sem uso do let ;; O que é esse 0.1? 

(defn valor-multa 
  [valor]
  (* valor 0.1))

;; Poderíamos colocar um comentário ; 0.1 representa o percentual da multa 

(defn valor-multa 
  [valor]
  (let [percentual-multa 0.1]
    (* valor percentual-multa)))

;; Calculando o valor total, sem o let 

(defn valor-total
  "Calcula valor total incluindo multa"
  [valor]
    (+ valor (* valor 0.1))) ; valor + valor da multa com percentual de 10%

;; O let é um vetor, podendo declarar vários símbolos e ;; o valor de um símbolo pode depender de um outro símbolo definido anteriormente. 

(defn valor-total
  "Calcula valor total incluindo multa"
  [valor]
  (let [percentual-multa 0.1
        valor-multa (* valor percentual-multa)]
    (+ valor valor-multa)))
